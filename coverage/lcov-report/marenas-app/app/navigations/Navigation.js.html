import React from 'react'
import { NavigationContainer } from "@react-navigation/native"
import { createBottomTabNavigator } from "@react-navigation/bottom-tabs"
import { Icon } from "react-native-elements"
 
import ReservarStack from "../navigations/ReservarStack"
import MisviajesStack from "../navigations/MisviajesStack"
import NotificacionesStack from "../navigations/NotificacionesStack"
import PerfilStack from "../navigations/PerfilStack"
 
const Tab = createBottomTabNavigator()
 
export default function Navigation() {
    return (
        <NavigationContainer>
            <Tab.Navigator
                initialRouteName="reservar"
                tabBarOptions={{
                    activeTintColor: "#f4809b",
                    inactiveTintColor: "#989898",
                    style: { zIndex: 1 }
                }}
                screenOptions={({ route }) => ({
                    tabBarIcon: ({ color }) => screenOptions(route, color)
                })}
            >
 
                <Tab.Screen name="reservar" component={ReservarStack} options={{ title: "Reservar" }} />
                <Tab.Screen name="misviajes" component={MisviajesStack} options={{ title: "Mis viajes" }} />
                <Tab.Screen name="notificaciones" component={NotificacionesStack} options={{ title: "Notificaciones" }} />
                <Tab.Screen name="perfil" component={PerfilStack} options={{ title: "Perfil" }} />
 
            </Tab.Navigator>
        </NavigationContainer>
 
    )
}
 
function screenOptions(route, color) {
    let iconName;
 
    switch (route.name) {
 
        case "reservar":
            iconName = "room-service-outline"
            break;
 
        case "misviajes":
            iconName = "airplane-takeoff"
            break;
        case "notificaciones":
            iconName = "flag-outline"
            break;
 
        case "reservar":
            iconName = "room-service-outline"
            break;
 
        case "perfil":
            iconName = "account-outline"
            break;
 
        default:
            break;
    }
    return (
        <Icon
 
            type="material-community"
            name={iconName}
            size={26}
            color={color !== "#989898" ? "white" : "#989898"}
            style={{
                backgroundColor: color !== "#989898" ? color : "transparent",
                borderRadius: 50,
                padding: 3,
                bottom: color !== "#989898" ? 10 : 0
            }}
 
        />
    )
 
}
